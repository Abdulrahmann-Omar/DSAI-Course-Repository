void bfs(vector<vector<int> >& adj, int startN, vector<bool>& visited)
{
    queue<int > q;

    visited[startN] = true;
    q.push(startN);

    while (!q.empty()) {

        int current = q.front();
        q.pop;
        cout << current << " ";


        for (int children : adj[current]) {

            if (!visited[children]) {

                visited[children] = true;
                q.push(children);
            }

        }

    }

    }

void addEdge(vector<vector<int> >& adj, int u, int v) {

    adj[u].push_back(v);
}
